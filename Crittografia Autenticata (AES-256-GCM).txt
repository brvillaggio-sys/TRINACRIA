import os
import base64
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.ciphers.aead import AESGCM

class SecureVault:
    def __init__(self, password: str):
        self.password = password.encode()

    def _derive_key(self, salt: bytes) -> bytes:
        """Deriva una chiave a 256-bit sicura dalla password usando PBKDF2."""
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=salt,
            iterations=480000, # Alto numero di iterazioni contro il brute-force
        )
        return kdf.derive(self.password)

    def encrypt_file(self, file_path: str, output_path: str):
        """Cifra un file garantendone confidenzialitŕ e integritŕ."""
        # 1. Genera un Salt (16 byte) e un Nonce (12 byte) casuali
        salt = os.urandom(16)
        nonce = os.urandom(12)
        
        # 2. Deriva la chiave e inizializza AES-GCM
        key = self._derive_key(salt)
        aesgcm = AESGCM(key)

        # 3. Leggi il file in chiaro
        with open(file_path, "rb") as f:
            data = f.read()

        # 4. Cifra i dati. Il Nonce DEVE essere univoco per ogni cifratura
        ciphertext = aesgcm.encrypt(nonce, data, None)

        # 5. Salva il file cifrato prepended con Salt e Nonce
        with open(output_path, "wb") as f:
            f.write(salt + nonce + ciphertext)
        
        print(f"[+] File '{file_path}' cifrato con successo in '{output_path}'.")

    def decrypt_file(self, encrypted_path: str, output_path: str):
        """Decifra e verifica l'integritŕ del file."""
        with open(encrypted_path, "rb") as f:
            file_data = f.read()

        # Estrai Salt (16 byte), Nonce (12 byte) e il resto sono i dati cifrati
        salt = file_data[:16]
        nonce = file_data[16:28]
        ciphertext = file_data[28:]

        key = self._derive_key(salt)
        aesgcm = AESGCM(key)

        try:
            # La decifratura GCM fallirŕ automaticamente se il file č stato alterato
            cleartext = aesgcm.decrypt(nonce, ciphertext, None)
            with open(output_path, "wb") as f:
                f.write(cleartext)
            print(f"[+] File decifrato e integritŕ verificata in '{output_path}'.")
        except Exception as e:
            print("[-] ERRORE CRITICO: Password errata o file corrotto/manomesso!")
